50  	public void setConsoleOutputMethod(final String loggingLevel) {
51  		try {
52  			// Try parsing to an integer
53  			int ordinalLevel = Integer.parseInt(loggingLevel);
54  			if (ordinalLevel > LoggingLevel.ERROR.getLevel() && ordinalLevel <= LoggingLevel.WARNING.getLevel()) {
55  				ApplicationLogger.setConsoleLoggingLevel(LoggingLevel.fromInteger(ordinalLevel));
56  				return;
57  			}
58  		} catch (IllegalArgumentException e) {
59  			// try getting the Level by name
60  			for (LoggingLevel level : LoggingLevel.values()) {
61  				if (level.name().equals(loggingLevel)) {
62  					ApplicationLogger.setConsoleLoggingLevel(level);
63  					return;
64  				}
65  			}
66  		}
67  
68  		throw new ParameterException(spec.commandLine(), loggingLevel + " Is not a valid logging level!");
69  	}

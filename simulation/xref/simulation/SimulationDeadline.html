7   import utility.game.step.IDeadline;

9   /**
10   * A {@link IDeadline } implementation with the possibility to reset the
11   * deadline. The lower and upper limit for the generated Deadline can be defined
12   * via cli-parameters.
13   */
14  public final class SimulationDeadline implements IDeadline {
15  
16      private ZonedDateTime deadline;
17      private final Random random;
18      private final int minSeconds;
19      private final int maxSeconds;
20  
21      private static int lowerTimeLimit = 2;
22      private static int upperTimeLimit = 15;
23  
24      /**
25       * Creates a new {@link SimulationDeadline} and calls
26       * {@link SimulationDeadline#resetDeadLine()}.
27       */
28      public SimulationDeadline() {
29          // Determine the min- and max-seconds, since the input via the command line does
30          // not guarantee that the lower limit is below the upper limit
31          minSeconds = Math.min(lowerTimeLimit, upperTimeLimit);
32          maxSeconds = Math.max(upperTimeLimit, lowerTimeLimit);
33  
34          random = new Random();
35          resetDeadLine();
36      }
37  
38      /**
39       * Resets the deadline by a random number of seconds between
40       * {@link SimulationDeadline#lowerTimeLimit} and
41       * {@link SimulationDeadline#upperTimeLimit}.
42       */
43      public void resetDeadLine() {
44          final int deadLineSeconds = random.nextInt(maxSeconds - minSeconds + 1) + minSeconds;
45          this.deadline = ZonedDateTime.now().plusSeconds(deadLineSeconds);
46      }
47  
48      @Override
49      public long getRemainingMilliseconds() {
50          final ZonedDateTime now = ZonedDateTime.now();
51          final Duration durationLeft = Duration.between(now, deadline);
52          return durationLeft.toMillis();
53      }
54  
55      /**
56       * Changes the global {@link SimulationDeadline#lowerTimeLimit} for all
57       * instances of {@link SimulationDeadline}.
58       * 
59       * @param lowerTimeLimit the minimum number of seconds for a random generated
60       *                       deadline
61       */
62      public static void setLowerTimeLimit(int lowerTimeLimit) {
63          SimulationDeadline.lowerTimeLimit = lowerTimeLimit;
64      }
65  
66      /**
67       * Changes the global {@link SimulationDeadline#upperTimeLimit} for all
68       * instances of {@link SimulationDeadline}.
69       * 
70       * @param upperTimeLimit the maximum number of seconds for a random generated
71       *                       deadline
72       */
73      public static void setUpperTimeLimit(int upperTimeLimit) {
74          SimulationDeadline.upperTimeLimit = upperTimeLimit;
75      }
